static QueueHandle_t s_free_queue = NULL;
static QueueHandle_t s_data_queue = NULL;

adxl345_buffer_t* get_filled_buffer(void) {
    adxl345_buffer_t *buf = NULL;
    if (xQueueReceive(s_data_queue, &buf, portMAX_DELAY) == pdTRUE) {
        return buf;
    }
    return NULL;
}

void release_buffer(adxl345_buffer_t *buf) {
    if (buf) {
        xQueueSend(s_free_queue, &buf, 0);
    }
}
